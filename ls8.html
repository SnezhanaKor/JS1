Не выполняя кода, ответить, что выведет браузер и почему: if (!("a" in window)) { var a = 1; } alert(a); //Если код будет запускаться в глобальном контексте, то браузер выведет "undefined", так как объявление var a существует и a будет считаться объявленной, но с неопределённым значением, так как тело внутри if не выполнится. Если же поместить этот кусок кода в локальный контекст внутри функции, то браузер выведет "1", так как будет считать a необъявленной и попадёт внутрь if. Если же var заменить на let, то будет ошибка выполнения, так как область видимости let a будет ограничена блоком внутри фигурных скобок. var b = function a(x) { x && a(--x); }; alert(a); //Код выдаст ошибку, так как переменная a не объявлена. function a(x) { return x * 2; } var a; alert(a); //Код выдаст ссылку на функцию a, которая объявлена ранее в глобальном контексте. Повторное объявление var a не меняет того, что это уже функция, в отличие от предыдущего примера, где в глобальном контексте объявлена ссылка на функцию b, а не функция a. function b(x, y, a) { arguments[2] = 10; alert(a); } b(1, 2, 3); //Код выдаст значение 10, так как она будет изменена через обращени к массиву аргументов. function a() { alert(this); } a.call(null); //Метод call() должен передавать в функцию null в качестве this. При проверке alert выводит объект window, то есть this указывает на текущее окно.
